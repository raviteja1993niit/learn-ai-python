App 1: agentic-ai-toggle-management — LangGraph / CrewAI / AutoGen

Current Architecture

1 orchestrator + 5 sub-agents using GitHub Copilot .github/agents markdown files in JetBrains IDE. Orchestrator routes plain-English commands; sub-agents handle Java scanning, report writing, Confluence lookup, decomposition planning, and workspace management.

LangGraph Improvement

Replace the orchestrator routing table with a LangGraph StateGraph. Each sub-agent becomes a node; conditional edges replace the manual routing table; built-in checkpointing handles long-running batch jobs automatically.

```python from typing import TypedDict, Annotated, List import operator from langgraph.graph import StateGraph, START, END from langgraph.checkpoint.sqlite import SqliteSaver

class ToggleState(TypedDict): command: str toggle_name: str repo: str registry: List[dict] analysis: List[dict] confluence_results: List[dict] report: str stage: str errors: Annotated[List[str], operator.add]

def route_command(state: ToggleState) -> str: cmd = state["command"].lower() if "analyze" in cmd: return "sa1_analysis" if "report" in cmd: return "sa2_report" if "confluence" in cmd: return "sa3_confluence" if "decomp" in cmd: return "sa4_advisor" if "rebuild registry" in cmd: return "sa1_registry" return "sa5_workspace"

def sa1_analysis(state: ToggleState) -> dict: """SA-1: Toggle Analysis Engine — scan Java repos, produce ON/OFF impact.""" # calls toggle-search-utility.ps1 via subprocess return {"stage": "analysis_complete", "analysis": [...]}

def sa2_report(state: ToggleState) -> dict: """SA-2: Report Writer — generate 6 report types.""" return {"stage": "report_complete", "report": "..."}

def sa3_confluence(state: ToggleState) -> dict: """SA-3: Confluence Enquiry — 4-tier search.""" return {"stage": "confluence_complete", "confluence_results": [...]}

def sa4_advisor(state: ToggleState) -> dict: """SA-4: Decomposition Advisor (on-request only).""" return {"stage": "advisor_complete"}

def sa5_workspace(state: ToggleState) -> dict: """SA-5: Workspace Manager — session state, onboarding, cleanup.""" return {"stage": "workspace_complete"}

Build graph

graph = StateGraph(ToggleState) graph.add_node("orchestrator", lambda s: s) # pure router graph.add_node("sa1_analysis", sa1_analysis) graph.add_node("sa1_registry", lambda s: {"stage": "registry_rebuilt"}) graph.add_node("sa2_report", sa2_report) graph.add_node("sa3_confluence", sa3_confluence) graph.add_node("sa4_advisor", sa4_advisor) graph.add_node("sa5_workspace", sa5_workspace)

graph.add_edge(START, "orchestrator") graph.add_conditional_edges("orchestrator", route_command, { "sa1_analysis": "sa1_analysis", "sa1_registry": "sa1_registry", "sa2_report": "sa2_report", "sa3_confluence": "sa3_confluence", "sa4_advisor": "sa4_advisor", "sa5_workspace": "sa5_workspace", }) for node in ["sa1_analysis","sa1_registry","sa2_report","sa3_confluence","sa4_advisor","sa5_workspace"]: graph.add_edge(node, END)

Checkpointing — survives crashes, supports resume

checkpointer = SqliteSaver.from_conn_string("toggle_sessions.db") app = graph.compile(checkpointer=checkpointer)

Run with thread_id for session isolation

### LangGraph CRAG (Corrective RAG) for App 2

```python
from langgraph.graph import StateGraph, END
from typing import TypedDict, List

class CRAGState(TypedDict):
    question: str
    documents: List[str]
    generation: str
    web_search_needed: bool

def retrieve(state):
    docs = retriever.invoke(state["question"])
    return {"documents": docs}

def grade_documents(state):
    """Grade each retrieved doc — if all irrelevant, trigger web search."""
    grading_prompt = "Is this document relevant to the question? Answer yes or no.\nDoc: {doc}\nQuestion: {question}"
    relevant = []
    for doc in state["documents"]:
        grade = llm.invoke(grading_prompt.format(doc=doc.page_content, question=state["question"]))
        if "yes" in grade.content.lower():
            relevant.append(doc)
    return {"documents": relevant, "web_search_needed": len(relevant) == 0}

def rewrite_query(state):
    """Rewrite query when retrieved docs are irrelevant."""
    new_q = llm.invoke(f"Rewrite this query to improve retrieval: {state['question']}")
    return {"question": new_q.content}

def generate_answer(state):
    context = "\n\n".join(d.page_content for d in state["documents"])
    answer = llm.invoke(f"Context:\n{context}\n\nQuestion: {state['question']}")
    return {"generation": answer.content}

def route_after_grading(state):
    return "rewrite" if state["web_search_needed"] else "generate"

crag = StateGraph(CRAGState)
crag.add_node("retrieve",        retrieve)
crag.add_node("grade_documents", grade_documents)
crag.add_node("rewrite",         rewrite_query)
crag.add_node("generate",        generate_answer)

crag.set_entry_point("retrieve")
crag.add_edge("retrieve", "grade_documents")
crag.add_conditional_edges("grade_documents", route_after_grading, {"rewrite": "rewrite", "generate": "generate"})
crag.add_edge("rewrite", "retrieve")
crag.add_edge("generate", END)

app = crag.compile()
result = app.invoke({"question": "What is the 3DS authentication flow?"})

2.1 PR Code Review Agent (GitHub MCP + LangGraph)

Problem: Manual PR reviews are slow, miss security issues, and provide inconsistent feedback.

Architecture:

GitHub PR (via MCP)
       │
       ▼
LangGraph StateGraph
  ├── fetch_pr_diff()       ← GitHub MCP: get changed files + diff
  ├── analyze_security()    ← LLM: OWASP Top 10, injection risks
  ├── analyze_quality()     ← LLM: code smells, complexity, naming
  ├── check_test_coverage() ← LLM: are new lines covered by tests?
  ├── compose_review()      ← LLM: merge all findings into review
  └── post_comment()        ← GitHub MCP: post review comment on PR

Tech stack: LangGraph + GitHub MCP server + GitHub Copilot LLM (free)

from langgraph.graph import StateGraph, END
from typing import TypedDict, List

class PRReviewState(TypedDict):
    pr_url: str
    diff: str
    files_changed: List[str]
    security_issues: List[str]
    quality_issues: List[str]
    test_gaps: List[str]
    final_review: str

def fetch_pr_diff(state):
    # Uses GitHub MCP tool
    diff = github_mcp.get_pull_request_diff(state["pr_url"])
    return {"diff": diff["diff"], "files_changed": diff["files"]}

def analyze_security(state):
    prompt = f"""Review this code diff for security vulnerabilities (OWASP Top 10, injection, secrets exposure):
{state['diff'][:8000]}
List each issue with severity (HIGH/MEDIUM/LOW) and line reference."""
    issues = llm.invoke(prompt)
    return {"security_issues": issues.content.split("\n")}

def analyze_quality(state):
    prompt = f"""Review this code diff for quality issues (code smells, complexity, naming, DRY violations):
{state['diff'][:8000]}
List specific actionable improvements."""
    issues = llm.invoke(prompt)
    return {"quality_issues": issues.content.split("\n")}

def compose_review(state):
    prompt = f"""Compose a professional, constructive PR review from these findings:
Security: {state['security_issues']}
Quality: {state['quality_issues']}
Format as GitHub markdown review with sections."""
    review = llm.invoke(prompt)
    return {"final_review": review.content}

def post_comment(state):
    github_mcp.post_review_comment(state["pr_url"], state["final_review"])
    return {}

graph = StateGraph(PRReviewState)
graph.add_node("fetch",    fetch_pr_diff)
graph.add_node("security", analyze_security)
graph.add_node("quality",  analyze_quality)
graph.add_node("compose",  compose_review)
graph.add_node("post",     post_comment)
graph.set_entry_point("fetch")
graph.add_edge("fetch",    "security")
graph.add_edge("fetch",    "quality")    # parallel!
graph.add_edge("security", "compose")
graph.add_edge("quality",  "compose")
graph.add_edge("compose",  "post")
graph.add_edge("post", END)

## 2.5 Automated Test Case Generator (LangGraph)

**Problem:** Writing unit tests is repetitive and often skipped under deadlines.

Source code file (Python / Java) │ ▼ LangGraph pipeline ├── parse_code() → extract functions/methods/classes ├── identify_cases() → for each function: happy path, edge cases, error cases ├── generate_tests() → write pytest/JUnit test code ├── validate_syntax() → parse AST to verify valid Python/Java ├── run_tests() → execute via subprocess └── fix_failures() → if tests fail, analyze error and fix (max 3 retries)


```python
class TestGenState(TypedDict):
    source_code: str
    language: str
    functions: List[dict]
    test_cases: List[dict]
    test_code: str
    test_results: str
    retry_count: int

def parse_code(state):
    prompt = f"Extract all public functions/methods from this {state['language']} code as JSON list [{{name, params, returns, docstring}}]:\n{state['source_code']}"
    functions = json.loads(llm.invoke(prompt).content)
    return {"functions": functions}

def generate_tests(state):
    prompt = f"""Write comprehensive pytest tests for these functions: {state['functions']}
Include: happy path, boundary values, error cases, type edge cases.
Return only valid Python code."""
    test_code = llm.invoke(prompt).content
    return {"test_code": test_code}

def run_tests(state):
    import subprocess
    Path("test_generated.py").write_text(state["test_code"])
    result = subprocess.run(["python", "-m", "pytest", "test_generated.py", "-v"],
                           capture_output=True, text=True)
    return {"test_results": result.stdout + result.stderr}

def should_retry(state):
    if "FAILED" in state["test_results"] and state["retry_count"] < 3:
        return "fix"
    return END

def fix_failures(state):
    prompt = f"These tests failed:\n{state['test_results']}\nOriginal tests:\n{state['test_code']}\nFix the failing tests."
    fixed = llm.invoke(prompt).content
    return {"test_code": fixed, "retry_count": state["retry_count"] + 1}

graph = StateGraph(TestGenState)
graph.add_node("parse",    parse_code)
graph.add_node("generate", generate_tests)
graph.add_node("run",      run_tests)
graph.add_node("fix",      fix_failures)
graph.set_entry_point("parse")
graph.add_edge("parse", "generate")
graph.add_edge("generate", "run")
graph.add_conditional_edges("run", should_retry, {"fix": "fix", END: END})
graph.add_edge("fix", "run")
app = graph.compile()

class MigrationState(TypedDict):
    file_path: str
    original_code: str
    migration_type: str  # "java8-to-21", "python2-to-3", "spring2-to-3"
    migrated_code: str
    validation_errors: List[str]
    diff: str

def detect_migration_needs(state):
    patterns = {
        "java8-to-21": ["new Thread(", "Collections.sort(", "StringBuffer", "@Override\n.*Date"],
        "spring2-to-3": ["javax.persistence", "javax.validation", "spring.datasource"],
    }
    for migration_type, pattern_list in patterns.items():
        if any(p in state["original_code"] for p in pattern_list):
            return {"migration_type": migration_type}
    return {"migration_type": "none"}

def migrate_code(state):
    if state["migration_type"] == "none":
        return {"migrated_code": state["original_code"]}
    prompt = f"""Migrate this {state['migration_type']} code. Apply ALL applicable modernizations.
Return ONLY the migrated code, no explanations.
Code: {state['original_code']}"""
    migrated = llm.invoke(prompt).content
    return {"migrated_code": migrated}

def generate_diff(state):
    import difflib
    diff = "\n".join(difflib.unified_diff(
        state["original_code"].splitlines(),
        state["migrated_code"].splitlines(),
        lineterm=""
    ))
    return {"diff": diff}

LangGraph is a library for building stateful, multi-actor applications with LLMs. Built on top of LangChain, it models workflows as directed graphs where nodes are Python functions (agents/tools) and edges define control flow. It adds persistent state, checkpointing, human-in-the-loop, and subgraph composition to LangChain's LCEL.

Core Concepts

```python from langgraph.graph import StateGraph, START, END from langgraph.checkpoint.sqlite import SqliteSaver from typing import TypedDict, Annotated, List import operator

1. Define State — shared across all nodes

class AgentState(TypedDict): messages: Annotated[List[str], operator.add] # list accumulates across nodes query: str documents: List[str] answer: str retry_count: int

2. Define Nodes — functions that update state

def retrieval_node(state: AgentState) -> dict: docs = retriever.invoke(state["query"]) return {"documents": [d.page_content for d in docs]}

def generation_node(state: AgentState) -> dict: context = "\n".join(state["documents"]) answer = llm.invoke(f"Context: {context}\nQuestion: {state['query']}").content return {"answer": answer, "messages": [answer]}

3. Conditional routing function

def should_retrieve(state: AgentState) -> str: if not state.get("documents"): return "retrieve" return "generate"

4. Build graph

graph = StateGraph(AgentState) graph.add_node("retrieve", retrieval_node) graph.add_node("generate", generation_node) graph.add_edge(START, "retrieve") graph.add_edge("retrieve", "generate") graph.add_edge("generate", END)

5. Compile with checkpointing

checkpointer = SqliteSaver.from_conn_string(":memory:") app = graph.compile(checkpointer=checkpointer)

6. Invoke with thread_id (session isolation)

config = {"configurable": {"thread_id": "session-123"}} result = app.invoke({"query": "What is RAG?", "messages": []}, config=config)

7. Stream intermediate steps

for event in app.stream({"query": "What is RAG?"}, config=config): for key, value in event.items(): print(f"Node: {key}, Output: {value}") ```

Human-in-the-Loop

```python

interrupt_before: pause graph before executing this node, let human review

app = graph.compile(checkpointer=checkpointer, interrupt_before=["generate"])

Run until interrupt

result = app.invoke({"query": "Delete all production data"}, config=config) print("Graph paused. Current state:", app.get_state(config))

Human reviews and approves

user_approval = input("Approve this action? (yes/no): ") if user_approval == "yes": # Resume from where it paused result = app.invoke(None, config=config) else: # Update state before resuming app.update_state(config, {"query": "Show production data stats instead"}) result = app.invoke(None, config=config) ```

Build a sub-graph

research_graph = StateGraph(ResearchState) research_graph.add_node("search", web_search) research_graph.add_node("summarize", summarize) research_graph.set_entry_point("search") research_graph.add_edge("search", "summarize") research_graph.add_edge("summarize", END) research_app = research_graph.compile()

Use subgraph as a node in parent graph

main_graph = StateGraph(MainState) main_graph.add_node("research", research_app) # subgraph as node! main_graph.add_node("write", write_report) main_graph.set_entry_point("research") main_graph.add_edge("research", "write") ```

Multi-Agent Supervisor Pattern

```python from langgraph.prebuilt import create_react_agent

Create sub-agents

research_agent = create_react_agent(llm, tools=[web_search, arxiv_search]) writer_agent = create_react_agent(llm, tools=[file_write]) coder_agent = create_react_agent(llm, tools=[python_repl, bash_tool])

class SupervisorState(TypedDict): messages: Annotated[list, operator.add] next: str

def supervisor(state): """Decide which agent to call next.""" options = ["research_agent", "writer_agent", "coder_agent", "FINISH"] prompt = f"Given conversation so far, who should act next? Options: {options}\nMessages: {state['messages']}" response = llm.invoke(prompt) return {"next": response.content.strip()}

def route_supervisor(state): return state["next"]

graph = StateGraph(SupervisorState) graph.add_node("supervisor", supervisor) graph.add_node("research_agent", lambda s: research_agent.invoke(s)) graph.add_node("writer_agent", lambda s: writer_agent.invoke(s)) graph.add_node("coder_agent", lambda s: coder_agent.invoke(s))

graph.set_entry_point("supervisor") graph.add_conditional_edges("supervisor", route_supervisor, { "research_agent": "research_agent", "writer_agent": "writer_agent", "coder_agent": "coder_agent", "FINISH": END }) for node in ["research_agent", "writer_agent", "coder_agent"]: graph.add_edge(node, "supervisor") # always return to supervisor ```

3.6 DSPy — Complete Reference

What is DSPy

DSPy (Stanford) replaces hand-crafted prompts with learnable modules. Instead of writing "You are an expert..." prompts manually, you define input/output signatures and DSPy's optimizer (Teleprompter) automatically discovers the best prompts/few-shot examples by evaluating on a small training set.

```python import dspy

Configure LLM

lm = dspy.LM("openai/gpt-4o-mini", base_url="...", api_key=token) dspy.configure(lm=lm)

1. Define signature (declarative: "what goes in, what comes out")

class ToggleImpactAnalysis(dspy.Signature): """Analyze the business impact of a feature toggle.""" code_context: str = dspy.InputField(desc="Java code snippet containing toggle usage") toggle_name: str = dspy.InputField(desc="Name of the feature toggle") on_impact: str = dspy.OutputField(desc="Business impact when toggle is ON") off_impact: str = dspy.OutputField(desc="Business impact when toggle is OFF") risk_level: str = dspy.OutputField(desc="Change risk: HIGH, MEDIUM, or LOW")

2. Use built-in modules

predictor = dspy.Predict(ToggleImpactAnalysis) # simple predict cot = dspy.ChainOfThought(ToggleImpactAnalysis) # chain-of-thought react = dspy.ReAct(ToggleImpactAnalysis, tools=[java_search_tool]) # agent

3. Build a pipeline (Module = composable unit)

class ToggleAnalysisPipeline(dspy.Module): def init(self): self.analyze = dspy.ChainOfThought(ToggleImpactAnalysis) self.summarize = dspy.Predict("impact_details -> executive_summary")

def forward(self, code_context, toggle_name):
    analysis = self.analyze(code_context=code_context, toggle_name=toggle_name)
    summary  = self.summarize(impact_details=f"ON: {analysis.on_impact}, OFF: {analysis.off_impact}")
    return dspy.Prediction(analysis=analysis, summary=summary.executive_summary)

pipeline = ToggleAnalysisPipeline()

4. Optimize prompts automatically using training examples

from dspy.teleprompt import BootstrapFewShot

def metric(example, prediction, trace=None): # Custom metric: does prediction match expected output? return example.on_impact.lower() in prediction.on_impact.lower()

train_set = [ dspy.Example(code_context="...", toggle_name="Enable-CIT", on_impact="CIT enabled in scheme", off_impact="CIT disabled").with_inputs("code_context", "toggle_name") ]

optimizer = BootstrapFewShot(metric=metric, max_bootstrapped_demos=3) optimized = optimizer.compile(pipeline, trainset=train_set) optimized.save("optimized_pipeline.json") # save for production

## Recipe 2: LangGraph CRAG (Corrective RAG)

```python
"""
Corrective RAG: if retrieved docs are irrelevant, rewrite query and retry.
Grade each doc → if all bad → rewrite query → retrieve again → generate.
"""
from langgraph.graph import StateGraph, END
from langchain_core.prompts import ChatPromptTemplate
from typing import TypedDict, List, Literal

class CRAGState(TypedDict):
    question: str
    documents: List[str]
    generation: str
    retry_count: int

GRADE_PROMPT = ChatPromptTemplate.from_template(
    "Is this document relevant to answering the question?\nDoc: {doc}\nQuestion: {question}\nAnswer only 'yes' or 'no'."
)
REWRITE_PROMPT = ChatPromptTemplate.from_template(
    "Rewrite this query to improve document retrieval. Be more specific.\nOriginal: {question}\nRewritten query:"
)

def retrieve(state):
    docs = hybrid_retriever.invoke(state["question"])
    return {"documents": [d.page_content for d in docs]}

def grade_documents(state):
    relevant = []
    for doc in state["documents"]:
        response = (GRADE_PROMPT | llm).invoke({"doc": doc[:500], "question": state["question"]})
        if "yes" in response.content.lower():
            relevant.append(doc)
    return {"documents": relevant}

def rewrite_query(state):
    response = (REWRITE_PROMPT | llm).invoke({"question": state["question"]})
    return {"question": response.content.strip(), "retry_count": state.get("retry_count", 0) + 1}

def generate(state):
    context  = "\n\n---\n\n".join(state["documents"])
    response = llm.invoke(f"Context:\n{context}\n\nAnswer this question: {state['question']}")
    return {"generation": response.content}

def route_after_grading(state) -> Literal["rewrite", "generate"]:
    if not state["documents"] and state.get("retry_count", 0) < 2:
        return "rewrite"
    return "generate"

graph = StateGraph(CRAGState)
graph.add_node("retrieve", retrieve)
graph.add_node("grade",    grade_documents)
graph.add_node("rewrite",  rewrite_query)
graph.add_node("generate", generate)
graph.set_entry_point("retrieve")
graph.add_edge("retrieve", "grade")
graph.add_conditional_edges("grade", route_after_grading, {"rewrite": "rewrite", "generate": "generate"})
graph.add_edge("rewrite", "retrieve")
graph.add_edge("generate", END)
app = graph.compile()

result = app.invoke({"question": "What is the 3DS authentication timeout value?", "retry_count": 0})
print(result["generation"])

    if repo == "all":

    else:
        file = base / f"toggle-registry-{repo}.csv"
    if not file.exists():
        return f"Registry file not found: {file}"
    return file.read_text(encoding="utf-8-sig")

@mcp.tool()
def search_toggle_usages(toggle_name: str, repo: str = "CPE") -> list[dict]:
    """
    Search a repository for toggle usages.
    Returns list of {file_path, line_number, code_snippet} dicts.
    """
    result = subprocess.run(
        ["powershell", "-File", "tools/active/toggle-search-utility.ps1", toggle_name, repo],
        capture_output=True, text=True, cwd="."
    )
    lines = result.stdout.strip().split("\n")
    usages = []
    for line in lines:
        if "|" in line:
            parts = line.split("|")
            if len(parts) >= 3:
                usages.append({"file": parts[0].strip(), "line": parts[1].strip(), "snippet": parts[2].strip()})
    return usages

@mcp.tool()
def get_analysis_status() -> dict:
    """Get current analysis progress: how many toggles analyzed vs total."""
    analysis_file = Path("data/analysis/toggle-codebase-usage-analysis.csv")
    input_file    = Path("data/enquiry/feature-toggle-confluence-page-enquiry.csv")
    if not analysis_file.exists():
        return {"analyzed": 0, "total": 0, "percentage": 0}
    analyzed = set()
    with open(analysis_file, encoding="utf-8-sig") as f:
        for row in csv.DictReader(f):
            analyzed.add(row.get("toggle_name","").strip())
    total = sum(1 for _ in open(input_file, encoding="utf-8-sig")) - 1
    return {"analyzed": len(analyzed), "total": total, "percentage": round(len(analyzed)/total*100, 1)}

Recipe 6: LangGraph Adaptive RAG (Route to Best Retrieval Strategy)

```python """ Adaptive RAG: classify query → route to best retrieval strategy. Technical acronyms → BM25; conceptual questions → vector; latest news → web search. """ from langgraph.graph import StateGraph, END from typing import TypedDict, Literal

class AdaptiveRAGState(TypedDict): question: str query_type: str # "technical" | "conceptual" | "current_events" documents: list answer: str

CLASSIFY_PROMPT = """Classify this query for optimal retrieval: - "technical": exact terms, acronyms, code, specifications (use BM25) - "conceptual": broad concepts, how-it-works questions (use vector) - "current_events": news, latest updates, 2024/2025 events (use web search)

Query: {question} Answer with one word only: technical | conceptual | current_events"""

def classify_query(state): response = llm.invoke(CLASSIFY_PROMPT.format(question=state["question"])) return {"query_type": response.content.strip().lower()}

def route_by_type(state) -> Literal["bm25_retrieve", "vector_retrieve", "web_retrieve"]: t = state["query_type"] if t == "technical": return "bm25_retrieve" if t == "conceptual": return "vector_retrieve" return "web_retrieve"

def bm25_retrieve(state): docs = bm25_retriever.invoke(state["question"]) return {"documents": [d.page_content for d in docs]}

def vector_retrieve(state): docs = vector_retriever.invoke(state["question"]) return {"documents": [d.page_content for d in docs]}

def web_retrieve(state): from langchain_community.tools import DuckDuckGoSearchRun search = DuckDuckGoSearchRun() result = search.invoke(state["question"]) return {"documents": [result]}

def generate(state): context = "\n\n".join(state["documents"]) response = llm.invoke(f"Answer based on context:\n{context}\n\nQuestion: {state['question']}") return {"answer": response.content}

graph = StateGraph(AdaptiveRAGState) graph.add_node("classify", classify_query) graph.add_node("bm25_retrieve", bm25_retrieve) graph.add_node("vector_retrieve",vector_retrieve) graph.add_node("web_retrieve", web_retrieve) graph.add_node("generate", generate)

graph.set_entry_point("classify") graph.add_conditional_edges("classify", route_by_type, { "bm25_retrieve": "bm25_retrieve", "vector_retrieve": "vector_retrieve", "web_retrieve": "web_retrieve", }) for node in ["bm25_retrieve", "vector_retrieve", "web_retrieve"]: graph.add_edge(node, "generate") graph.add_edge("generate", END)
