Build a Local Code Assistant App

# 1. Pull a code model
ollama pull qwen2.5-coder:7b

# 2. Install dependencies
pip install openai rich

# 3. code_assistant.py
from openai import OpenAI
from rich.console import Console
from rich.markdown import Markdown

client  = OpenAI(base_url="http://localhost:11434/v1", api_key="ollama")
console = Console()

SYSTEM = """You are an expert software engineer.
When asked to write code, produce clean, well-commented code.
Always explain what the code does after writing it."""

history = [{"role": "system", "content": SYSTEM}]

console.print("[bold blue]Local Code Assistant[/] (qwen2.5-coder • Ctrl+C to quit)\n")
while True:
    user_input = console.input("[bold green]You:[/] ")
    history.append({"role": "user", "content": user_input})

    response = client.chat.completions.create(
        model="qwen2.5-coder:7b",
        messages=history,
        stream=True
    )
    assistant_text = ""
    console.print("[bold yellow]Assistant:[/]")
    for chunk in response:
        delta = chunk.choices[0].delta.content or ""
        assistant_text += delta
        print(delta, end="", flush=True)
    print()
    console.print(Markdown(assistant_text))
    history.append({"role": "assistant", "content": assistant_text})

# Requirements: 4–8 GB GPU RAM (or CPU with patience)
pip install diffusers transformers accelerate torch torchvision
from diffusers import StableDiffusionPipeline, DPMSolverMultistepScheduler
import torch

MODEL_ID = "stabilityai/stable-diffusion-2-1"

pipe = StableDiffusionPipeline.from_pretrained(
    MODEL_ID,
    torch_dtype=torch.float16   # use float32 for CPU
)
pipe.scheduler = DPMSolverMultistepScheduler.from_config(pipe.scheduler.config)
pipe = pipe.to("cuda")          # change to "cpu" if no GPU

prompt = "A futuristic software developer workspace, neon lights, cyberpunk style, 4K"
negative_prompt = "blurry, low quality, watermark, text"

image = pipe(
    prompt,
    negative_prompt=negative_prompt,
    num_inference_steps=25,     # more steps = better quality but slower
    guidance_scale=7.5
).images[0]

image.save("output.png")
print("Image saved to output.png")

Local Vision App Example

import ollama, base64, sys
from pathlib import Path

# Pull first: ollama pull minicpm-v
def analyse_image(image_path: str, question: str = "Describe this image in detail") -> str:
    img_data = Path(image_path).read_bytes()
    img_b64  = base64.b64encode(img_data).decode()

    response = ollama.chat(
        model="minicpm-v",
        messages=[{
            "role": "user",
            "content": question,
            "images": [img_b64]
        }]
    )
    return response["message"]["content"]

# Usage
result = analyse_image("screenshot.png", "List all the text visible in this image")
print(result)
# Vision with GitHub Copilot Free API (GPT-4o Vision — no local GPU needed)
import subprocess, base64
from openai import OpenAI
from pathlib import Path

token  = subprocess.run(["gh","auth","token"], capture_output=True, text=True).stdout.strip()
client = OpenAI(base_url="https://models.inference.ai.azure.com", api_key=token)

def analyse_image_copilot(image_path: str, question: str) -> str:
    img_b64 = base64.b64encode(Path(image_path).read_bytes()).decode()
    resp = client.chat.completions.create(
        model="gpt-4o",
        messages=[{"role":"user","content":[
            {"type":"text","text": question},
            {"type":"image_url","image_url":{"url": f"data:image/png;base64,{img_b64}"}}
        ]}]
    )
    return resp.choices[0].message.content

print(analyse_image_copilot("chart.png", "What trend does this chart show?"))

Local Audio Pipeline App

# Full voice assistant pipeline: Mic → Whisper → LLM → TTS → Speaker
pip install faster-whisper sounddevice numpy openai kokoro soundfile
import sounddevice as sd
import numpy as np
import subprocess
from faster_whisper import WhisperModel
from openai import OpenAI
from kokoro import KPipeline
import soundfile as sf
import tempfile, os

# Setup
stt_model  = WhisperModel("base", device="cpu", compute_type="int8")
tts_pipe   = KPipeline(lang_code="a")
gh_token   = subprocess.run(["gh","auth","token"], capture_output=True, text=True).stdout.strip()
llm_client = OpenAI(base_url="https://models.inference.ai.azure.com", api_key=gh_token)

def record_audio(seconds=5, sample_rate=16000) -> np.ndarray:
    print(f"Recording for {seconds}s...")
    audio = sd.rec(int(seconds * sample_rate), samplerate=sample_rate,
                   channels=1, dtype='float32')
    sd.wait()
    return audio.flatten()

def transcribe(audio: np.ndarray) -> str:
    with tempfile.NamedTemporaryFile(suffix=".wav", delete=False) as f:
        sf.write(f.name, audio, 16000)
        segments, _ = stt_model.transcribe(f.name)
        os.unlink(f.name)
    return " ".join(s.text for s in segments).strip()

def ask_llm(text: str) -> str:
    resp = llm_client.chat.completions.create(
        model="gpt-4o-mini",
        messages=[{"role":"user","content": text}]
    )
    return resp.choices[0].message.content

def speak(text: str):
    chunks = [c for c, _, _ in tts_pipe(text, voice="af_heart")]
    audio  = np.concatenate(chunks)
    sd.play(audio, 24000); sd.wait()

# Run the pipeline
audio   = record_audio(seconds=5)
user_q  = transcribe(audio)
print(f"You said: {user_q}")
answer  = ask_llm(user_q)
print(f"Assistant: {answer}")
speak(answer)

14. 🎤 Q&A

What is the difference between a transformer and a diffusion model?

Transformers are sequence-to-sequence architectures based on self-attention. They process tokens in parallel and learn relationships between all positions. Used for: text LLMs (GPT, LLaMA, Claude), vision transformers (ViT), code models, and embedding models. They generate by predicting the next token autoregressively.

Diffusion models work by gradually adding Gaussian noise to an image (forward process) and learning to reverse this (denoising). At inference, they start from pure noise and iteratively denoise guided by a text prompt (via CLIP cross-attention). Used for: image generation (SD, FLUX, DALL-E 3), video generation, audio generation. They generate by iterative denoising over N steps (10–50 typically).

Modern multimodal models combine both: DALL-E 3 uses a transformer to convert the prompt into a rich description, then a diffusion model generates the image.

When should I use a local model vs an API?
  • Use local (Ollama) when: data privacy is required, offline capability needed, no API budget, experimenting without rate limits, or processing large volumes where API costs would be high
  • Use API when: you need the best quality (GPT-4o, Claude 3.5), don't have a GPU, need image generation (DALL-E 3), need multimodal capability not available locally, or building a product where inference speed matters more than cost
  • Hybrid approach: use free API tiers (Groq, Gemini, GitHub Copilot) for development and testing, switch to paid APIs or local models for production based on cost/quality requirements
What is quantisation and why does it matter for local models?

Quantisation reduces the precision of model weights from float32 (32-bit) or float16 (16-bit) to int8 (8-bit) or int4 (4-bit). This reduces:

  • Memory usage: A 70B model at float16 needs ~140 GB RAM. At 4-bit (Q4_K_M GGUF), it needs ~40 GB — runnable on 2× RTX 3090s or a Mac Studio M2 Ultra
  • Inference speed: Smaller data types compute faster on CPU/GPU

Quality tradeoff: Q4 loses ~2–3% on benchmarks vs float16. Q8 loses <1%. For most chat/RAG use cases, Q4 is indistinguishable from full precision. Ollama automatically uses Q4_K_M quantisation for most models.

How do I choose the right embedding model for my RAG app?

Key factors: (1) Domain — general models (all-MiniLM) work for most tasks; specialised models exist for code (CodeBERT), medical (BiomedBERT), legal text. (2) Context length — all-MiniLM truncates at 256 tokens; nomic-embed-text handles 8K tokens (better for long documents). (3) Multilingual — use multilingual-e5-large or Cohere embed-multilingual-v3 for non-English content. (4) Speed vs quality — MiniLM-L6 is 5× faster than MPNet but scores ~2% lower on MTEB. (5) Cost — for high-volume production, local models (free) beat OpenAI at scale. Start with nomic-embed-text via Ollama — it's free, local, 8K context, and scores 62+ on MTEB.

What free resources can I use to experiment without a GPU?
  • Text LLMs: Groq (free API, 600 tokens/sec), Google AI Studio (free Gemini), GitHub Copilot Free API, OpenRouter free models
  • Image generation: HuggingFace Spaces (SDXL, FLUX in browser), Playground AI (free tier), Adobe Firefly (free tier)
  • GPU compute (training/fine-tuning): Google Colab (free T4), Kaggle (free P100, 30 hrs/week), Lightning AI (free A10G), HuggingFace ZeroGPU
  • Local CPU models: Ollama with 3B–7B models runs surprisingly well on a modern CPU (8–10 tokens/sec for llama3.2:3b on M1 Mac / Intel i7)
What is the role of CLIP in image generation models?

CLIP (Contrastive Language–Image Pretraining by OpenAI) is a model trained to align text and image representations in a shared embedding space. In diffusion models like Stable Diffusion:

  1. Your text prompt is passed through the CLIP text encoder → produces a text embedding
  2. This embedding is used as cross-attention conditioning in the U-Net denoising network
  3. At each denoising step, the U-Net is "guided" toward images that match the text embedding
  4. Guidance scale controls how strongly the image follows the prompt (7.5 = balanced; 15 = very strict but may lose quality)

FLUX.1 replaced CLIP with T5-XXL (a more powerful text encoder) — which is why FLUX handles complex, detailed prompts much better than SD 1.5.
